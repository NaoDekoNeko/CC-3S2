namespace ReSOSgame
{
    public class Controller
    {
        public Tablero Board { get; set; }

        public Game Game { get; set; }

        public Player Player1 { get; set; }

        public Player Player2 { get; set; }

        public Player CurrentPlayer
        {
            get => Board.JugadorActual;
            set => Board.JugadorActual = value;
        }

        public Tablero.Jugador Turn
        {
            get => Board.Turno;
        }

        public Tablero.Cell Ficha
        {
            get => CurrentPlayer.Ficha;
        }

        public int X => CurrentPlayer.X;
        public int Y => CurrentPlayer.Y;
        public int Tamanio => Board.Tamanio;

        public void InitTurn()
        {
            // Nota: Antes de usar el init turn debe haberse inicializado el player1
            CurrentPlayer = Player1;
        }

        public void ChangeTurn()
        {
            CurrentPlayer = Board.Turno == Tablero.Jugador.JAZUL ? Player1 : Player2;
        }
    }
}
