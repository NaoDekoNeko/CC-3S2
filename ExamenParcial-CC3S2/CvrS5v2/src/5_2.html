using System;
using static ReSOSgame.Board;

namespace ReSOSgame
{
    public class Consola
    {
        private readonly Board board;

        public Consola(Board board)
        {
            this.board = board;
        }

        public void DisplayBoard()
        {
            for (var row = 0; row < board.Tamanio; row++)
            {
                Console.WriteLine("-------");
                for (var column = 0; column < board.Tamanio; column++)
                    Console.Write("|" + Symbol(board.GetCell(row, column)));
                Console.WriteLine("|");
            }

            Console.WriteLine("-------");
        }

        private char Symbol(Cell cell)
        {
            if (cell == Cell.S)
                return 'S';
            if (cell == Cell.O)
                return 'O';
            return ' ';
        }
    }
}
