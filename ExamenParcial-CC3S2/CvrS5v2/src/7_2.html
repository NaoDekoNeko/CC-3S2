using static ReSOSgame.Board;

namespace ReSOSgame
{
    public class ScoreValidator
    {
        private readonly Board board;

        public ScoreValidator(Board board)
        {
            this.board = board;
        }

        // Funcion booleana que verifica si el board esta lleno
        public bool FullBoard()
        {
            for (var i = 0; i < board.Tamanio; i++)
            for (var j = 0; j < board.Tamanio; j++)
                if (board.GetCell(i, j) == Cell.VACIA)
                    return false;
            return true;
        }

        public bool HasOnePoint(int row, int col, Cell ficha)
        {
            switch (ficha)
            {
                case Cell.S:
                    return (board.GetCell(row + 1, col + 1) == Cell.O && board.GetCell(row + 2, col + 2) == Cell.S) ||
                           (board.GetCell(row - 1, col - 1) == Cell.O && board.GetCell(row - 2, col - 2) == Cell.S) ||
                           (board.GetCell(row, col + 1) == Cell.O && board.GetCell(row, col + 2) == Cell.S) ||
                           (board.GetCell(row, col - 1) == Cell.O && board.GetCell(row, col - 2) == Cell.S) ||
                           (board.GetCell(row + 1, col) == Cell.O && board.GetCell(row + 2, col) == Cell.S) ||
                           (board.GetCell(row - 1, col) == Cell.O && board.GetCell(row - 2, col) == Cell.S);
                case Cell.O:
                    return (board.GetCell(row - 1, col - 1) == Cell.S && board.GetCell(row + 1, col + 1) == Cell.S) ||
                           (board.GetCell(row + 1, col - 1) == Cell.S && board.GetCell(row - 1, col + 1) == Cell.S) ||
                           (board.GetCell(row - 1, col) == Cell.S && board.GetCell(row + 1, col) == Cell.S) ||
                           (board.GetCell(row, col - 1) == Cell.S && board.GetCell(row, col + 1) == Cell.S);
                case Cell.VACIA:
                case Cell.INVALIDA:
                default:
                    return false;
            }
        }

        // Funcion que verifica si el game se ha terminado
        public bool GameOver()
        {
            return board.EstadoDeJuego != GameState.JUGANDO;
        }
    }
}
